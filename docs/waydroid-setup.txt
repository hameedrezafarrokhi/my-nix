  # Manual Setup (Recommended)

   # Initial Step

    # 1.Gui: Just lauch waydroid

    # 2.CLI:
     # sudo waydroid init           (Vanilla)
     # sudo waydroid init -f        (Force)
     # sudo waydroid init -s GAPPS  (Google play)

     # sudo waydroid container start

     # waydroid session start
     # waydroid session stop

   # Authenticate PC for Google Play
     # https://docs.waydro.id/faq/google-play-certification

   # IF You Want To Delete Residual Files OR For Reinitialization:
     # sudo rm -rf /var/lib/waydroid /home/.waydroid ~/waydroid ~/.share/waydroid ~/.local/share/applications/*aydroid* ~/.local/share/waydroid

   # To Check If Working:
     # sudo systemctl status waydroid-container

   # To Update The Android Image:
     # sudo waydroid upgrade

   # ARM Translation (https://github.com/casualsnek/waydroid_script):
     # sudo waydroid-script
     ## (Select Android 11 >> Install >> libhoudini...)
